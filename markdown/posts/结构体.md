结构体是用户自己定义的数据类型，它所占的空间是所有成员的空间之和
但是 ，每个成员的占用空间数是 4 的倍数

形如：
struct 结构体名
{
  具体成员
};
定义结构体成员

    ```c
    #include <stdio.h>
    struct Tongxue
    {
        char name[80];
        int age;
        float fenshu;
    }; //末尾需要添加分号
    //定义一个结构体命名为Tongxue
    int main(void)
    {
        struct Tongxue  people;
        printf("请输入同学名字 : ");
        scanf("%s", people.name);
        //用结构体名.具体成员名，就可以访问具体成员
        printf("请输入同学年龄 : ");
        scanf("%d", &people.age);
        printf("请输入同学名字 : ");
        scanf("%f", &people.fenshu);
        printf("你好，%s 同学今年 %d岁，考了%f分", people.name, people.age, people.fenshu);
        //打印输出
        return 0;
    }
    ```

tips ： 定义的时候也可以直接把名字放结构体成员的话括弧后面如
struct 结构体名
{
  具体成员
} name ;

这样就是全局变量

初始值
我们可以直接用大括号指定成员的初始值

    ```c
    #include <stdio.h>
    struct Tongxue
    {
        char name[80];
        int age;
        float fenshu;
    }; 
    int main(void)
    {
        struct Tongxue people =
            {
                "gejing",
                100,
                66.6
            };
        //直接初始化结构体
        printf("你好，%s 同学今年 %d岁，考了%f分", people.name, people.age, people.fenshu);
        return 0;
    }
    ```
我们也可以直接指定某个成员的初始值

    ```c
    #include <stdio.h>
    struct Tongxue
    {
        char name[80];
        int age;
        float fenshu;
    }; 
    int main(void)
    {
        struct Tongxue  people={.age = 100};
        //指定成员中的一位的初始值

        printf("请输入同学名字 : ");
        scanf("%s", people.name);
        printf("请输入同学分数 : ");
        scanf("%f", &people.fenshu);
        printf("你好，%s 同学今年 %d岁，考了%f分", people.name, people.age, people.fenshu);
        return 0;
    }

    ```

结构体里面嵌套结构体

    ```c
    #include <stdio.h>
    struct Date
    {
        int year;
        int month;
        int day;
    };
    struct Tongxue
    {
        char name[80];
        struct Date date;
        //嵌套结构体
        float fenshu;
    }mate; 
    int main(void)
    {
        printf("请输入同学的名字: ");
        scanf("%s", mate.name);
        printf("请输入同学的出生日期: ");
        scanf("%d%d%d", &mate.date.year,&mate.date.month,&mate.date.day);
        //通过  结构体名.结构体名.成员名访问
        printf("请输入同学的分数: ");
        scanf("%f", &mate.fenshu);
        printf("你好，%s 同学出生于%d年%d月%d日，考了%f分", mate.name, mate.date.year,mate.date.month,mate.date.day,mate.fenshu);
        return 0;
    }

    ```
结构体数组 ： 直接在名字后加上长度就行了

    ```c
    #include <stdio.h>
    struct Date
    {
        int year;
        int month;
        int day;
    };
    struct Tongxue
    {
        char name[80];
        struct Date date;
        float fenshu;
    }mate[5]; 
    //直接定义5个结构体变量
    int main(void)
    {
        for (int i = 0; i < 5;i++)
        {
            printf("请输入第%d个同学的名字: ",i+1);
            scanf("%s", mate[i].name);
            //通过下标引用结构体
            printf("请输入该同学的出生日期: ");
            scanf("%d%d%d", &mate[i].date.year,&mate[i].date.month,&mate[i].date.day);
            printf("请输入该同学的分数: ");
            scanf("%f", &mate[i].fenshu);
        }
        for (int i = 0; i < 5;i++)
        {
            printf("你好，%s 同学出生于%d年%d月%d日，考了%f分\n", mate[i].name, mate[i].date.year,mate[i].date.month,mate[i].date.day,mate[i].fenshu);
        }
        return 0;
    }

    ```
结构体指针 struct 结构体名 *变量名
基本上和常规类型一样的用法

    ```c
    #include <stdio.h>
    struct Name
    {
        char x[80];
    };

    void show(struct Name *p,int x)
    //定义结构体指针
    {
        for (int i = 0; i < x;i++)
        {
            printf("%s\n", p->x);
            p++;
        }
    }

    int main(void)
    {
        int x;
        printf("一共有多少个同学？  ");
        scanf("%d",&x);
        struct Name classMate[x];
        printf("请依次输入同学的姓名\n ");
        for (int i = 0; i < x;i++)
        {
            scanf("%s",classMate[i].x);
        }
        show(classMate, x);
        //直接传入地址打印
        return 0;
    }
    ```

我们如果不喜欢 每次都打struct的话还可以用这个方法定义

    ```c
    #include <stdio.h>
    typedef struct 
    {
        int x;
    } Name;
    //直接声明结构体名字
    //可以声明多个但是意义不大
    int main(void)
    {
        int x;
        Name t;
        //直接定义
        scanf("%d",&x);
        t.x = x;
        printf("%d mmm ",t.x);
        return 0;
    }

    ```
